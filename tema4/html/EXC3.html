Metoda Directa

function [ y ] = MetDirecta2( X,Y,x )
    syms P;
    n=length(X)-1;
    Pn=0;
    for i=1:n+1
        for j=1:n+1
            A(i,j)=X(i)^(j-1);
        end
    end
    a=A\transpose(Y);
    for i=1:length(a)
        Pn=Pn+a(i)*(P^(i-1));
    end
    Pn=matlabFunction(Pn,'vars',{P});
    y=Pn(x);
end


Metoda Lagrange

function [ y ] = MetLagrange(X, Y, x)
    n=length(X)-1;
    syms P;
    Pn=0;
    for k=1:n+1
        Lnk=1;
        for i=1:n+1
            if i~=k
                Lnk=Lnk*((P-X(i)) / (X(k)-X(i)));
            end
        end
        Pn=Pn + Lnk*Y(k);
    end
    Pn=matlabFunction(Pn, 'vars' , {P});
    y=Pn(x);
end

Metoda Newton

function [ y ] = MetNew(X,Y,x)
    n=length(X)-1;
    syms P;
    Pn=0;
    for i=1:n+1
        for j=1:n+1
            if j==1
                A(i,j)=1;
            elseif j>i
                A(i,j)=0;
            else 
                p=1;
                for k=1:j-1
                    p=p*(X(i)-X(k));
                end
                A(i,j)=p;
            end
        end
    end
    C = SubsAsc(A, Y');
    for i=1:n+1
        co=C(i);
        for j=1:(i-1)
            co=co*(P-X(j));
        end
        Pn=Pn+co;
    end
    Pn=matlabFunction(Pn, 'vars' , {P});
    y=Pn(x);
end

Substitutie Ascendenta

function [ x ] = SubsAsc( A, b )
n=length(b);
x(1)=(1/A(1,1))*b(1);
    for k=2:n
        s=0;
        for j=1:k-1
            s=s+A(k,j)*x(j);
        end
        x(k)=(b(k)-s)/A(k,k);
    end
    x=transpose(x);
end
